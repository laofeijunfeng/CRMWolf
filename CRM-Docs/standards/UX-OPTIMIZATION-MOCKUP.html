代码实现建议（在 Customers.vue 中添加）:
<!-- 已筛选汇总区 - 在 filter-tabs-bar 后添加 -->
<div v-if="hasActiveFilters" class="active-filter-summary">
  <div class="filter-count-badge">
    <span class="badge-num">{{ activeFilterCount }}</span>
    <span class="badge-text">个筛选条件</span>
  </div>
  <div class="filter-tags-row">
    <el-tag v-for="(value, field) in filterValues"
            :key="field" closable @close="handleFilterClear(field)">
      {{ getFilterLabel(field) }}: {{ getFilterValuePreview(value) }}
    </el-tag>
  </div>
  <el-button text size="small" @click="handleReset">清除全部</el-button>
</div>

<script>
const hasActiveFilters = computed(() => Object.keys(filterValues.value).length > 0)
const activeFilterCount = computed(() => Object.keys(filterValues.value).length)

const getFilterLabel = (field: string) => {
  const labels: Record<string, string> = {
    account_name: '客户名称',
    city: '城市',
    status: '状态',
    score: '热力值'
  }
  return labels[field] || field
}

const getFilterValuePreview = (value: FilterValue) => {
  if (value.search) return value.search
  if (value.select) {
    const statusLabels = { 0: '跟进中', 1: '已赢单', 2: '已输单', 3: '已失效' }
    return statusLabels[value.select as number] || value.select
  }
  return '...'
}
</script>
